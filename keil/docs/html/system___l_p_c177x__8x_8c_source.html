46 //
47 // <e> Clock Configuration
48 // <h> System Controls and Status Register (SCS - address 0x400F C1A0)
49 // <o1.0> EMC Shift Control Bit
50 // <i> Controls how addresses are output on the EMC address pins for static memories
51 // <0=> Static CS addresses match bus width; AD[1] = 0 for 32 bit, AD[0] = 0 for 16+32 bit (Bit 0 is 0)
52 // <1=> Static CS addresses start at LSB 0 regardless of memory width (Bit 0 is 1)
53 //
54 // <o1.1> EMC Reset Disable Bit
55 // <i> If 0 (zero), all registers and functions of the EMC are initialized upon any reset condition
56 // <i> If 1, EMC is still retained its state through a warm reset
57 // <0=> Both EMC resets are asserted when any type of chip reset event occurs (Bit 1 is 0)
58 // <1=> Portions of EMC will only be reset by POR or BOR event (Bit 1 is 1)
59 //
60 // <o1.2> EMC Burst Control
61 // <i> Set to 1 to prevent multiple sequential accesses to memory via EMC static memory chip selects
62 // <0=> Burst enabled (Bit 2 is 0)
63 // <1=> Bust disbled (Bit 2 is 1)
64 //
65 // <o1.3> MCIPWR Active Level
66 // <i> Selects the active level for the SD card interface signal SD_PWR
67 // <0=> SD_PWR is active low (inverted output of the SD Card interface block) (Bit 3 is 0)
68 // <1=> SD_PWR is active high (follows the output of the SD Card interface block) (Bit 3 is 1)
69 //
70 // <o1.4> Main Oscillator Range Select
71 // <0=> In Range 1 MHz to 20 MHz (Bit 4 is 0)
72 // <1=> In Range 15 MHz to 25 MHz (Bit 4 is 1)
73 //
74 // <o1.5> Main Oscillator enable
75 // <i> 0 (zero) means disabled, 1 means enable
76 //
77 // <o1.6> Main Oscillator status (Read-Only)
78 // </h>
79 //
80 // <h> Clock Source Select Register (CLKSRCSEL - address 0x400F C10C)
81 // <o2.0> CLKSRC: Select the clock source for sysclk to PLL0 clock
82 // <0=> Internal RC oscillator (Bit 0 is 0)
83 // <1=> Main oscillator (Bit 0 is 1)
84 // </h>
85 //
86 // <e3>PLL0 Configuration (Main PLL PLL0CFG - address 0x400F C084)
87 // <i> F_in is in the range of 1 MHz to 25 MHz
88 // <i> F_cco = (F_in * M * 2 * P) is in range of 156 MHz to 320 MHz
89 // <i> PLL out clock = (F_cco / (2 * P)) is in rane of 9.75 MHz to 160 MHz
90 //
91 // <o4.0..4> MSEL: PLL Multiplier Value
92 // <i> M Value
93 // <1-32><#-1>
94 //
95 // <o4.5..6> PSEL: PLL Divider Value
96 // <i> P Value
97 // <0=> 1
98 // <1=> 2
99 // <2=> 4
100 // <3=> 8
101 // </e>
102 //
103 // <e5>PLL1 Configuration (Alt PLL PLL1CFG - address 0x400F C0A4)
104 // <i> F_in is in the range of 1 MHz to 25 MHz
105 // <i> F_cco = (F_in * M * 2 * P) is in range of 156 MHz to 320 MHz
106 // <i> PLL out clock = (F_cco / (2 * P)) is in rane of 9.75 MHz to 160 MHz
107 //
108 // <o6.0..4> MSEL: PLL Multiplier Value
109 // <i> M Value
110 // <1-32><#-1>
111 //
112 // <o6.5..6> PSEL: PLL Divider Value
113 // <i> P Value
114 // <0=> 1
115 // <1=> 2
116 // <2=> 4
117 // <3=> 8
118 // </e>
119 //
120 // <h> CPU Clock Selection Register (CCLKSEL - address 0x400F C104)
121 // <o7.0..4> CCLKDIV: Select the value for divider of CPU clock (CCLK)
122 // <i> 0: The divider is turned off. No clock will be provided to the CPU
123 // <i> n: The input clock is divided by n to produce the CPU clock
124 // <0-31>
125 //
126 // <o7.8> CCLKSEL: Select the input to the divider of CPU clock
127 // <0=> sysclk clock is used
128 // <1=> Main PLL0 clock is used
129 // </h>
130 //
131 // <h> USB Clock Selection Register (USBCLKSEL - 0x400F C108)
132 // <o8.0..4> USBDIV: USB clock (source PLL0) divider selection
133 // <0=> Divider is off and no clock provides to USB subsystem
134 // <1=> Divider value is 1 (The source clock is divided by 1)
135 // <4=> Divider value is 4 (The source clock is divided by 4)
136 // <6=> Divider value is 6 (The source clock is divided by 6)
137 //
138 // <o8.8..9> USBSEL: Select the source for USB clock divider
139 // <i> When CPU clock is selected, the USB can be accessed
140 // <i> by software but cannot perform USB functions
141 // <0=> sysclk clock (the clock input to PLL0)
142 // <1=> The clock output from PLL0
143 // <2=> The clock output from PLL1
144 // </h>
145 //
146 // <h> EMC Clock Selection Register (EMCCLKSEL - address 0x400F C100)
147 // <o9.0> EMCDIV: Set the divider for EMC clock
148 // <0=> Divider value is 1
149 // <1=> Divider value is 2 (EMC clock is equal a half of input clock)

151 //

369 void SystemCoreClockUpdate (void) /* Get Core Clock Frequency */
370 {
371  /* Determine clock frequency according to clock register values */
372  if ((LPC_SC->CCLKSEL &0x100) == 0) { /* cclk = sysclk */
373  if ((LPC_SC->CLKSRCSEL & 0x01) == 0) { /* sysclk = irc_clk */
374  SystemCoreClock = __CLK_DIV(IRC_OSC , (LPC_SC->CCLKSEL & 0x1F));
375  PeripheralClock = __CLK_DIV(IRC_OSC , (LPC_SC->PCLKSEL & 0x1F));
376  EMCClock = (IRC_OSC / ((LPC_SC->EMCCLKSEL & 0x01)+1));
377  }
378  else { /* sysclk = osc_clk */
379  if ((LPC_SC->SCS & 0x40) == 0) {
380  SystemCoreClock = 0; /* this should never happen! */
381  PeripheralClock = 0;
382  EMCClock = 0;
383  }
384  else {
385  SystemCoreClock = __CLK_DIV(OSC_CLK , (LPC_SC->CCLKSEL & 0x1F));
386  PeripheralClock = __CLK_DIV(OSC_CLK , (LPC_SC->PCLKSEL & 0x1F));
387  EMCClock = (OSC_CLK / ((LPC_SC->EMCCLKSEL & 0x01)+1));
388  }
389  }
390  }
391  else { /* cclk = pll_clk */
392  if ((LPC_SC->PLL0STAT & 0x100) == 0) { /* PLL0 not enabled */
393  SystemCoreClock = 0; /* this should never happen! */
394  PeripheralClock = 0;
395  EMCClock = 0;
396  }
397  else {
398  if ((LPC_SC->CLKSRCSEL & 0x01) == 0) { /* sysclk = irc_clk */
399  SystemCoreClock = __CLK_DIV(IRC_OSC * ((LPC_SC->PLL0STAT & 0x1F) + 1) , (LPC_SC->CCLKSEL & 0x1F));
400  PeripheralClock = __CLK_DIV(IRC_OSC * ((LPC_SC->PLL0STAT & 0x1F) + 1) , (LPC_SC->PCLKSEL & 0x1F));
401  EMCClock = (IRC_OSC * ((LPC_SC->PLL0STAT & 0x1F) + 1) / ((LPC_SC->EMCCLKSEL & 0x01)+1));
402  }
403  else { /* sysclk = osc_clk */
404  if ((LPC_SC->SCS & 0x40) == 0) {
405  SystemCoreClock = 0; /* this should never happen! */
406  PeripheralClock = 0;
407  EMCClock = 0;
408  }
409  else {
410  SystemCoreClock = __CLK_DIV(OSC_CLK * ((LPC_SC->PLL0STAT & 0x1F) + 1) , (LPC_SC->CCLKSEL & 0x1F));
411  PeripheralClock = __CLK_DIV(OSC_CLK * ((LPC_SC->PLL0STAT & 0x1F) + 1) , (LPC_SC->PCLKSEL & 0x1F));
412  EMCClock = (OSC_CLK * ((LPC_SC->PLL0STAT & 0x1F) + 1) / ((LPC_SC->EMCCLKSEL & 0x01)+1));
413  }
414  }
415  }
416  }
417  /* ---update USBClock------------------*/
418  if(LPC_SC->USBCLKSEL & (0x01<<8))//Use PLL0 as the input to the USB clock divider
419  {
420  switch (LPC_SC->USBCLKSEL & 0x1F)
421  {
422  case 0:
423  USBClock = 0; //no clock will be provided to the USB subsystem
424  break;
425  case 4:
426  case 6:
427  if(LPC_SC->CLKSRCSEL & 0x01) //pll_clk_in = main_osc
428  USBClock = (OSC_CLK * ((LPC_SC->PLL0STAT & 0x1F) + 1) / (LPC_SC->USBCLKSEL & 0x1F));
429  else //pll_clk_in = irc_clk
430  USBClock = (IRC_OSC * ((LPC_SC->PLL0STAT & 0x1F) + 1) / (LPC_SC->USBCLKSEL & 0x1F));
431  break;
432  default:
433  USBClock = 0; /* this should never happen! */
434  }
435  }
436  else if(LPC_SC->USBCLKSEL & (0x02<<8))//usb_input_clk = alt_pll (pll1)
437  {
438  if(LPC_SC->CLKSRCSEL & 0x01) //pll1_clk_in = main_osc
439  USBClock = (OSC_CLK * ((LPC_SC->PLL1STAT & 0x1F) + 1));
440  else //pll1_clk_in = irc_clk
441  USBClock = (IRC_OSC * ((LPC_SC->PLL0STAT & 0x1F) + 1));
442  }
443  else
444  USBClock = 0; /* this should never happen! */
445 }
